# 集成测试文档

## 1. 概述

集成测试用于验证不同模块之间的交互是否正常工作。本文档描述了UI设计管理系统的集成测试方案。

**测试工具：**
- Supertest：HTTP请求测试
- Jest：测试框架
- MySQL Test Database：测试数据库

### 2.3 数据库测试工具

创建数据库测试工具（backend/test-utils/db.js）

```javascript
import mysql2 from 'mysql2/promise';
import { readFileSync } from 'fs';
import { fileURLToPath } from 'url';
import { dirname, join } from 'path';

const __filename = fileURLToPath(import.meta.url);
const __dirname = dirname(__filename);

/**
 * 创建测试数据库连接
 */
export const createTestDbConnection = async () => {
  const connection = await mysql2.createConnection({
    host: process.env.DB_HOST,
    port: process.env.DB_PORT,
    user: process.env.DB_USER,
    password: process.env.DB_PASSWORD,
    database: process.env.DB_NAME,
    multipleStatements: true
  });

  return connection;
};

/**
 * 初始化测试数据库
 */
export const initTestDb = async () => {
  const connection = await createTestDbConnection();
  
  try {
    // 读取并执行SQL脚本
    const sql = readFileSync(
      join(__dirname, '../config/database.test.sql'),
      'utf-8'
    );
    
    await connection.query(sql);
  } finally {
    await connection.end();
  }
};

/**
 * 清空测试数据库
 */
export const clearTestDb = async () => {
  const connection = await createTestDbConnection();
  
  try {
    await connection.query('DELETE FROM users');
    await connection.query('ALTER TABLE users AUTO_INCREMENT = 1');
  } finally {
    await connection.end();
  }
};

/**
 * 关闭数据库连接
 */
export const closeTestDb = async (connection) => {
  if (connection) {
    await connection.end();
  }
};
```

## 3. API集成测试

### 3.1 认证流程集成测试

#### 测试文件：backend/__tests__/integration/authFlow.test.js

```javascript
import request from 'supertest';
import express from 'express';
import { initTestDb, clearTestDb } from '../../test-utils/db.js';
import bcrypt from 'bcryptjs';
import jwt from 'jsonwebtoken';
import authRoutes from '../../routes/authRoutes.js';

describe('认证流程集成测试', () => {
  let app;
  let token;

  beforeAll(async () => {
    // 初始化测试数据库
    await initTestDb();

    // 创建Express应用
    app = express();
    app.use(express.json());
    app.use('/api', authRoutes);
  });

  afterAll(async () => {
    // 清理测试数据库
    await clearTestDb();
  });

  describe('完整的认证流程', () => {
    test('用户注册 -> 登录 -> 获取用户信息 -> 退出登录', async () => {
      // 1. 注册新用户
      const registerResponse = await request(app)
        .post('/api/register')
        .send({
          username: 'integration_test',
          password: 'test123',
          nickname: '集成测试用户'
        });

      expect(registerResponse.status).toBe(201);
      expect(registerResponse.body.code).toBe(201);
      expect(registerResponse.body.data.username).toBe('integration_test');

      // 2. 使用新注册的账号登录
      const loginResponse = await request(app)
        .post('/api/login')
        .send({
          username: 'integration_test',
          password: 'test123'
        });

      expect(loginResponse.status).toBe(200);
      expect(loginResponse.body.code).toBe(200);
      expect(loginResponse.body.data).toHaveProperty('token');
      expect(loginResponse.body.data).toHaveProperty('userInfo');

      token = loginResponse.body.data.token;
      const userId = loginResponse.body.data.userInfo.id;

      // 3. 使用token获取用户信息
      const userInfoResponse = await request(app)
        .get('/api/user/info')
        .set('Authorization', `Bearer ${token}`);

      expect(userInfoResponse.status).toBe(200);
      expect(userInfoResponse.body.code).toBe(200);
      expect(userInfoResponse.body.data.id).toBe(userId);
      expect(userInfoResponse.body.data.username).toBe('integration_test');
      expect(userInfoResponse.body.data).not.toHaveProperty('password');

      // 4. 退出登录
      const logoutResponse = await request(app)
        .post('/api/logout')
        .set('Authorization', `Bearer ${token}`);

      expect(logoutResponse.status).toBe(200);
      expect(logoutResponse.body.code).toBe(200);
      expect(logoutResponse.body.message).toBe('退出成功');
    });

    test('使用预设账号登录', async () => {
      // 使用数据库中预设的admin账号登录
      const loginResponse = await request(app)
        .post('/api/login')
        .send({
          username: 'admin',
          password: '123456'
        });

      expect(loginResponse.status).toBe(200);
      expect(loginResponse.body.code).toBe(200);
      expect(loginResponse.body.data.userInfo.username).toBe('admin');
    });
  });

  describe('Token验证流程', () => {
    let validToken;
    let expiredToken;

    beforeAll(async () => {
      // 生成有效token
      validToken = jwt.sign(
        { id: 1, username: 'admin', nickname: '管理员' },
        process.env.JWT_SECRET,
        { expiresIn: '1h' }
      );

      // 生成过期token
      expiredToken = jwt.sign(
        { id: 1, username: 'admin', nickname: '管理员' },
        process.env.JWT_SECRET,
        { expiresIn: '-1h' } // 已过期
      );
    });

    test('有效token可以访问受保护接口', async () => {
      const response = await request(app)
        .get('/api/user/info')
        .set('Authorization', `Bearer ${validToken}`);

      expect(response.status).toBe(200);
      expect(response.body.code).toBe(200);
    });

    test('过期token应该被拒绝', async () => {
      const response = await request(app)
        .get('/api/user/info')
        .set('Authorization', `Bearer ${expiredToken}`);

      expect(response.status).toBe(403);
      expect(response.body.code).toBe(403);
      expect(response.body.message).toContain('无效或已过期');
    });

    test('无token应该被拒绝', async () => {
      const response = await request(app)
        .get('/api/user/info');

      expect(response.status).toBe(401);
      expect(response.body.code).toBe(401);
      expect(response.body.message).toBe('未提供访问令牌');
    });

    test('错误的token格式应该被拒绝', async () => {
      const response = await request(app)
        .get('/api/user/info')
        .set('Authorization', 'InvalidFormat');

      expect(response.status).toBe(401);
      expect(response.body.code).toBe(401);
    });
  });

  describe('错误处理流程', () => {
    test('重复注册应该失败', async () => {
      // 第一次注册
      await request(app)
        .post('/api/register')
        .send({
          username: 'duplicate_test',
          password: 'test123',
          nickname: '重复测试'
        });

      // 第二次注册相同用户名
      const response = await request(app)
        .post('/api/register')
        .send({
          username: 'duplicate_test',
          password: 'test123',
          nickname: '重复测试'
        });

      expect(response.status).toBe(400);
      expect(response.body.code).toBe(400);
      expect(response.body.message).toBe('用户名已存在');
    });

    test('错误的用户名或密码应该登录失败', async () => {
      const response = await request(app)
        .post('/api/login')
        .send({
          username: 'nonexistent_user',
          password: 'wrong_password'
        });

      expect(response.status).toBe(401);
      expect(response.body.code).toBe(401);
      expect(response.body.message).toBe('用户名或密码错误');
    });

    test('参数验证失败应该返回400', async () => {
      const response = await request(app)
        .post('/api/login')
        .send({
          username: 'ab', // 用户名太短
          password: '123' // 密码太短
        });

      expect(response.status).toBe(400);
      expect(response.body.code).toBe(400);
      expect(response.body.message).toBe('请求参数验证失败');
      expect(response.body.data).toBeInstanceOf(Array);
      expect(response.body.data.length).toBeGreaterThan(0);
    });
  });
});
```

### 3.2 数据库集成测试

#### 测试文件：backend/__tests__/integration/database.test.js

```javascript
import { initTestDb, clearTestDb, createTestDbConnection } from '../../test-utils/db.js';
import bcrypt from 'bcryptjs';

describe('数据库集成测试', () => {
  let connection;

  beforeAll(async () => {
    await initTestDb();
    connection = await createTestDbConnection();
  });

  afterAll(async () => {
    await clearTestDb();
    await connection.end();
  });

  beforeEach(async () => {
    // 每个测试前清空表
    await connection.query('DELETE FROM users');
    await connection.query('ALTER TABLE users AUTO_INCREMENT = 1');
  });

  test('应该能够插入用户', async () => {
    const hashedPassword = await bcrypt.hash('test123', 10);

    await connection.query(
      'INSERT INTO users (username, password, nickname, avatar) VALUES (?, ?, ?, ?)',
      ['testuser', hashedPassword, '测试用户', 'https://example.com/avatar.png']
    );

    const [users] = await connection.query(
      'SELECT * FROM users WHERE username = ?',
      ['testuser']
    );

    expect(users.length).toBe(1);
    expect(users[0].username).toBe('testuser');
    expect(users[0].nickname).toBe('测试用户');
    expect(users[0].password).toBe(hashedPassword);
  });

  test('应该能够查询用户', async () => {
    const hashedPassword = await bcrypt.hash('test123', 10);

    await connection.query(
      'INSERT INTO users (username, password, nickname) VALUES (?, ?, ?)',
      ['testuser', hashedPassword, '测试用户']
    );

    const [users] = await connection.query(
      'SELECT id, username, nickname FROM users WHERE username = ?',
      ['testuser']
    );

    expect(users.length).toBe(1);
    expect(users[0].username).toBe('testuser');
    expect(users[0]).not.toHaveProperty('password'); // 查询时不返回密码
  });

  test('应该能够更新用户最后登录时间', async () => {
    const hashedPassword = await bcrypt.hash('test123', 10);

    const [result] = await connection.query(
      'INSERT INTO users (username, password, nickname) VALUES (?, ?, ?)',
      ['testuser', hashedPassword, '测试用户']
    );

    const userId = result.insertId;

    await connection.query(
      'UPDATE users SET last_login_time = NOW() WHERE id = ?',
      [userId]
    );

    const [users] = await connection.query(
      'SELECT last_login_time FROM users WHERE id = ?',
      [userId]
    );

    expect(users[0].last_login_time).not.toBeNull();
  });

  test('应该能够删除用户', async () => {
    const hashedPassword = await bcrypt.hash('test123', 10);

    await connection.query(
      'INSERT INTO users (username, password, nickname) VALUES (?, ?, ?)',
      ['testuser', hashedPassword, '测试用户']
    );

    await connection.query('DELETE FROM users WHERE username = ?', ['testuser']);

    const [users] = await connection.query(
      'SELECT * FROM users WHERE username = ?',
      ['testuser']
    );

    expect(users.length).toBe(0);
  });

  test('用户名应该唯一', async () => {
    const hashedPassword = await bcrypt.hash('test123', 10);

    await connection.query(
      'INSERT INTO users (username, password, nickname) VALUES (?, ?, ?)',
      ['testuser', hashedPassword, '测试用户']
    );

    await expect(
      connection.query(
        'INSERT INTO users (username, password, nickname) VALUES (?, ?, ?)',
        ['testuser', hashedPassword, '另一个用户']
      )
    ).rejects.toThrow();
  });

  test('密码应该加密存储', async () => {
    const plainPassword = 'test123';
    const hashedPassword = await bcrypt.hash(plainPassword, 10);

    await connection.query(
      'INSERT INTO users (username, password, nickname) VALUES (?, ?, ?)',
      ['testuser', hashedPassword, '测试用户']
    );

    const [users] = await connection.query(
      'SELECT password FROM users WHERE username = ?',
      ['testuser']
    );

    expect(users[0].password).not.toBe(plainPassword); // 密码不是明文
    expect(users[0].password).toBe(hashedPassword); // 密码是加密后的hash

    // 验证bcrypt可以正确验证密码
    const isValid = await bcrypt.compare(plainPassword, users[0].password);
    expect(isValid).toBe(true);
  });
});
```

### 4.2 E2E测试用例

#### 测试文件：e2e/auth.spec.js

```javascript
import { test, expect } from '@playwright/test';

test.describe('认证功能E2E测试', () => {
  test.beforeEach(async ({ page }) => {
    // 每个测试前导航到登录页
    await page.goto('/');
  });

  test('应该能够成功登录', async ({ page }) => {
    // 输入用户名
    await page.fill('#username', 'admin');
    
    // 输入密码
    await page.fill('#password', '123456');
    
    // 点击登录按钮
    await page.click('.login-button');
    
    // 等待跳转到首页
    await page.waitForURL('/home');
    
    // 验证页面标题
    await expect(page.locator('.home-header h2')).toHaveText('首页');
    
    // 验证用户信息显示
    await expect(page.locator('#username-display')).toBeVisible();
  });

  test('应该显示登录错误', async ({ page }) => {
    // 输入错误的密码
    await page.fill('#username', 'admin');
    await page.fill('#password', 'wrongpassword');
    
    // 点击登录按钮
    await page.click('.login-button');
    
    // 等待错误提示出现
    await expect(page.locator('.error-message')).toBeVisible();
    
    // 验证错误消息
    await expect(page.locator('.error-message')).toContainText('失败');
  });

  test('表单验证应该正常工作', async ({ page }) => {
    // 不输入任何内容直接点击登录
    await page.click('.login-button');
    
    // 浏览器应该显示原生验证提示
    const usernameInput = page.locator('#username');
    await expect(usernameInput).toHaveAttribute('required');
    
    const passwordInput = page.locator('#password');
    await expect(passwordInput).toHaveAttribute('required');
  });

  test('记住密码功能应该正常工作', async ({ page }) => {
    // 勾选记住密码
    await page.check('#remember-password');
    
    // 输入凭据
    await page.fill('#username', 'admin');
    await page.fill('#password', '123456');
    
    // 登录
    await page.click('.login-button');
    await page.waitForURL('/home');
    
    // 退出登录
    await page.click('.logout-button');
    await page.waitForURL('/');
    
    // 刷新页面
    await page.reload();
    
    // 验证输入框是否自动填充
    await expect(page.locator('#username')).toHaveValue('admin');
    await expect(page.locator('#password')).toHaveValue('123456');
    await expect(page.locator('#remember-password')).toBeChecked();
  });

  test('密码显示/隐藏切换应该正常工作', async ({ page }) => {
    const passwordInput = page.locator('#password');
    const toggleButton = page.locator('.toggle-password');
    
    // 输入密码
    await passwordInput.fill('123456');
    
    // 默认应该是隐藏的
    await expect(passwordInput).toHaveAttribute('type', 'password');
    
    // 点击切换按钮
    await toggleButton.click();
    
    // 应该显示明文
    await expect(passwordInput).toHaveAttribute('type', 'text');
    
    // 再次点击
    await toggleButton.click();
    
    // 应该恢复隐藏
    await expect(passwordInput).toHaveAttribute('type', 'password');
  });
});

test.describe('首页功能E2E测试', () => {
  test.beforeEach(async ({ page }) => {

  test('应该能够退出登录', async ({ page }) => {
    // 点击退出按钮

    await page.waitForURL('/');
    await expect(page.locator('.login-page')).toBeVisible();

  test('移动端应该能够打开侧边栏', async ({ page, isMobile }) => {
    if (!isMobile) {
      test.skip();
      return;
    }
    
    const menuButton = page.locator('.menu-button');
    const sideNav = page.locator('.side-nav');
    const overlay = page.locator('.side-nav-overlay');
    
    // 点击菜单按钮
    await menuButton.click();
    
    // 验证侧边栏显示
    await expect(sideNav).toBeVisible();
    await expect(overlay).toBeVisible();
    
    // 点击遮罩层关闭
    await overlay.click();
    
    // 验证侧边栏隐藏
    await expect(sideNav).not.toBeVisible();
  });
});
```

## 7. 测试数据管理

### 7.1 测试数据清理

创建测试数据清理脚本（backend/test-utils/cleanup.js）

```javascript
import { createTestDbConnection } from './db.js';

/**
 * 清理所有测试数据
 */
export const cleanupTestData = async () => {
  const connection = await createTestDbConnection();
  
  try {
    // 清空所有表
    const tables = await connection.query('SHOW TABLES');
    
    for (const table of tables[0]) {
      const tableName = Object.values(table)[0];
      await connection.query(`DELETE FROM ${tableName}`);
      await connection.query(`ALTER TABLE ${tableName} AUTO_INCREMENT = 1`);
    }
  } finally {
    await connection.end();
  }
};
```

## 9. 测试最佳实践

1. **隔离测试环境**
   - 使用独立的测试数据库
   - 每个测试前清理数据
   - 避免测试之间的依赖

2. **使用真实HTTP请求**
   - 集成测试应该使用真实的HTTP请求
   - 使用Supertest模拟HTTP客户端
   - 测试完整的请求-响应周期

3. **模拟外部依赖**
   - Mock第三方API调用
   - Mock文件系统操作
   - Mock定时器

4. **测试边界条件**
   - 空值、null、undefined
   - 极端数值
   - 特殊字符
   - 并发场景

5. **保持测试独立性**
   - 每个测试应该独立运行
   - 不要依赖测试的执行顺序
   - 使用beforeEach/afterEach清理
