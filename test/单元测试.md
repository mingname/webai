本文档描述了UI设计管理系统的单元测试方案，包括后端控制器测试、中间件测试、前端Store测试和API测试。

- 后端：Jest + Supertest
- 前端：Vitest

- 分支覆盖率：≥ 75%
- 函数覆盖率：≥ 85%

import request from 'supertest';
import express from 'express';
import { login, register, getUserInfo, logout } from '../controllers/authController.js';
import db from '../config/database.js';
import bcrypt from 'bcryptjs';

// Mock数据库
jest.mock('../config/database.js');
jest.mock('bcryptjs');

describe('Auth Controller - 登录功能', () => {
  let app;

  beforeAll(() => {
    app = express();
    app.use(express.json());
    app.post('/login', login);
    app.post('/register', register);
    app.get('/user/info', getUserInfo);
    app.post('/logout', logout);
  });

  afterEach(() => {
    jest.clearAllMocks();
  });

  describe('POST /login', () => {
    test('应该成功登录并返回token', async () => {
      // Mock数据库查询返回用户
      const mockUser = {
        id: 1,
        username: 'admin',
        password: 'hashed_password',
        nickname: '管理员',
        avatar: 'https://example.com/avatar.png',
        email: null
      };
      
      db.query.mockResolvedValueOnce([[mockUser]]);
      db.query.mockResolvedValueOnce([]); // 更新登录时间
      bcrypt.compare.mockResolvedValueOnce(true);

      const response = await request(app)
        .post('/login')
        .send({
          username: 'admin',
          password: '123456'
        });

      expect(response.status).toBe(200);
      expect(response.body.code).toBe(200);
      expect(response.body.message).toBe('登录成功');
      expect(response.body.data).toHaveProperty('token');
      expect(response.body.data).toHaveProperty('userInfo');
      expect(response.body.data.userInfo).not.toHaveProperty('password');
      expect(db.query).toHaveBeenCalledTimes(2);
      expect(bcrypt.compare).toHaveBeenCalledWith('123456', 'hashed_password');
    });

    test('用户名不存在时应该返回401错误', async () => {
      db.query.mockResolvedValueOnce([[]]);

      const response = await request(app)
        .post('/login')
        .send({
          username: 'nonexistent',
          password: '123456'
        });

      expect(response.status).toBe(401);
      expect(response.body.code).toBe(401);
      expect(response.body.message).toBe('用户名或密码错误');
      expect(response.body.data).toBeNull();
    });

    test('密码错误时应该返回401错误', async () => {
      const mockUser = {
        id: 1,
        username: 'admin',
        password: 'hashed_password'
      };
      
      db.query.mockResolvedValueOnce([[mockUser]]);
      bcrypt.compare.mockResolvedValueOnce(false);

      const response = await request(app)
        .post('/login')
        .send({
          username: 'admin',
          password: 'wrongpassword'
        });

      expect(response.status).toBe(401);
      expect(response.body.code).toBe(401);
      expect(response.body.message).toBe('用户名或密码错误');
    });

    test('数据库错误时应该返回500错误', async () => {
      db.query.mockRejectedValueOnce(new Error('Database error'));

      const response = await request(app)
        .post('/login')
        .send({
          username: 'admin',
          password: '123456'
        });

      expect(response.status).toBe(500);
      expect(response.body.code).toBe(500);
      expect(response.body.message).toBe('服务器内部错误');
    });
  });

  describe('POST /register', () => {
    test('应该成功注册新用户', async () => {
      // Mock用户名检查（不存在）
      db.query.mockResolvedValueOnce([[]]);
      // Mock密码加密
      bcrypt.hash.mockResolvedValueOnce('hashed_password');
      // Mock插入用户
      db.query.mockResolvedValueOnce([{ insertId: 3 }]);

      const response = await request(app)
        .post('/register')
        .send({
          username: 'newuser',
          password: '123456',
          nickname: '新用户'

      expect(response.status).toBe(201);
      expect(response.body.code).toBe(201);

      expect(response.body.data).toHaveProperty('username', 'newuser');
      expect(response.body.data).toHaveProperty('nickname', '新用户');
      expect(response.body.data).toHaveProperty('avatar');
      expect(db.query).toHaveBeenCalledTimes(2);
      expect(bcrypt.hash).toHaveBeenCalledWith('123456', 10);
    });

    test('用户名已存在时应该返回400错误', async () => {
      db.query.mockResolvedValueOnce([[{ id: 1 }]]);

      const response = await request(app)
        .post('/register')
        .send({
          username: 'admin',

          nickname: '管理员'
        });

      expect(response.status).toBe(400);
      expect(response.body.code).toBe(400);
      expect(response.body.message).toBe('用户名已存在');

      db.query.mockResolvedValueOnce([[]]);
      bcrypt.hash.mockResolvedValueOnce('hashed_password');
      db.query.mockResolvedValueOnce([{ insertId: 3 }]);

      const response = await request(app)
        .post('/register')
        .send({
          username: 'newuser',
          password: '123456'
        });

      expect(response.body.data.nickname).toBe('newuser');
      expect(db.query).toHaveBeenCalledWith(
        expect.any(String),
        ['newuser', 'hashed_password', 'newuser', expect.any(String)]

    });
  });

  describe('GET /user/info', () => {
    test('应该成功获取用户信息', async () => {
      const mockUser = {
        id: 1,
        username: 'admin',
        nickname: '管理员',
        avatar: 'https://example.com/avatar.png',
        email: 'admin@example.com',
        phone: '13800138000',
        created_at: '2026-01-26T00:00:00.000Z'
      };
      
      // Mock req.user
      const req = { user: { id: 1 } };
      db.query.mockResolvedValueOnce([[mockUser]]);

      // 直接调用控制器函数
      const res = {
        json: jest.fn(),
        status: jest.fn().mockReturnThis()
      };

      await getUserInfo(req, res);

      expect(res.json).toHaveBeenCalledWith({
        code: 200,
        message: '获取成功',
        data: mockUser
      });
    });

    test('用户不存在时应该返回404错误', async () => {
      const req = { user: { id: 999 } };
      db.query.mockResolvedValueOnce([[]]);

      const res = {
        json: jest.fn(),
        status: jest.fn().mockReturnThis()
      };

      await getUserInfo(req, res);

      expect(res.status).toHaveBeenCalledWith(404);
      expect(res.json).toHaveBeenCalledWith({
        code: 404,
        message: '用户不存在',
        data: null

    });
  });

  describe('POST /logout', () => {
    test('应该成功退出登录', async () => {

      expect(res.json).toHaveBeenCalledWith({

import jwt from 'jsonwebtoken';
import { authenticateToken, generateToken } from '../middleware/auth.js';

jest.mock('jsonwebtoken');

describe('Auth Middleware - JWT认证', () => {
  const mockSecret = 'test-secret';
  const mockUser = {
    id: 1,
    username: 'admin',
    nickname: '管理员'
  };

  beforeEach(() => {
    process.env.JWT_SECRET = mockSecret;
  });

  afterEach(() => {
    jest.clearAllMocks();
  });

  describe('generateToken', () => {
    test('应该生成有效的JWT token', () => {
      const mockToken = 'valid.jwt.token';
      jwt.sign.mockReturnValue(mockToken);

      const token = generateToken(mockUser);

      expect(token).toBe(mockToken);
      expect(jwt.sign).toHaveBeenCalledWith(
        {
          id: mockUser.id,
          username: mockUser.username,
          nickname: mockUser.nickname
        },
        mockSecret,
        { expiresIn: '24h' }
      );
    });

    test('应该使用环境变量中的过期时间', () => {
      process.env.JWT_EXPIRES_IN = '7d';
      const mockToken = 'valid.jwt.token';
      jwt.sign.mockReturnValue(mockToken);

      generateToken(mockUser);

      expect(jwt.sign).toHaveBeenCalledWith(
        expect.any(Object),
        mockSecret,
        { expiresIn: '7d' }
      );
    });
  });

  describe('authenticateToken', () => {
    let req, res, next;

    beforeEach(() => {
      req = {
        headers: {}
      };
      res = {
        status: jest.fn().mockReturnThis(),
        json: jest.fn()
      };
      next = jest.fn();
    });

    test('缺少token时应该返回401错误', async () => {
      await authenticateToken(req, res, next);

      expect(res.status).toHaveBeenCalledWith(401);
      expect(res.json).toHaveBeenCalledWith({
        code: 401,
        message: '未提供访问令牌',
        data: null
      });
      expect(next).not.toHaveBeenCalled();
    });

    test('有效的token应该通过认证', async () => {
      const mockToken = 'valid.jwt.token';
      req.headers.authorization = `Bearer ${mockToken}`;
      
      jwt.verify.mockImplementation((token, secret, callback) => {
        callback(null, mockUser);
      });

      await authenticateToken(req, res, next);

      expect(req.user).toEqual(mockUser);
      expect(next).toHaveBeenCalled();
      expect(res.status).not.toHaveBeenCalled();
    });

    test('无效的token应该返回403错误', async () => {
      const mockToken = 'invalid.jwt.token';
      req.headers.authorization = `Bearer ${mockToken}`;
      
      jwt.verify.mockImplementation((token, secret, callback) => {
        callback(new Error('Invalid token'), null);
      });

      await authenticateToken(req, res, next);

      expect(res.status).toHaveBeenCalledWith(403);
      expect(res.json).toHaveBeenCalledWith({
        code: 403,
        message: '访问令牌无效或已过期',
        data: null
      });
      expect(next).not.toHaveBeenCalled();
    });

    test('Authorization头格式错误时应该返回401错误', async () => {
      req.headers.authorization = 'InvalidFormat';

      await authenticateToken(req, res, next);

      expect(res.status).toHaveBeenCalledWith(401);
      expect(res.json).toHaveBeenCalledWith({
        code: 401,
        message: '未提供访问令牌',
        data: null
      });
    });
  });
});

import { setActivePinia, createPinia } from 'pinia';

## 7. 测试最佳实践

1. **测试命名规范**
   - 使用描述性的测试名称
   - 遵循"应该..."的命名模式
   - 中文或英文都可以，但要保持一致

2. **测试隔离**
   - 每个测试应该独立运行
   - 使用beforeEach/afterEach清理状态
   - 避免测试之间的依赖

3. **Mock使用**
   - 只mock外部依赖
   - 不要mock被测试的代码
   - 验证mock函数的调用

4. **覆盖率目标**
   - 不要为了覆盖率而写测试
   - 优先测试核心业务逻辑
   - 关注边界情况和错误处理

5. **持续集成**
   - 每次提交都运行测试
   - 测试失败时阻止合并
   - 定期检查覆盖率报告
